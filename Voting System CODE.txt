    // Password check for buttons
    function askPassword(action) {
      const userInput = prompt("Enter Password:");
      if (userInput === PASSWORD) {
        if (action === "results") {
          toggleResults();
        } else if (action === "reset") {
          resetVotes();
        }
      } else {
        alert("Wrong Password! Please Try Agian!");
      }
    }

    // Toggle show/hide results
    function toggleResults() {
      const box = document.getElementById("resultsBox");
      const btn = document.querySelector(".toggle-btn");

      if (box.style.display === "none") {
        updateResults();
        box.style.display = "block";
        btn.innerText = "Hide Results";
      } else {
        box.style.display = "none";
        btn.innerText = "Show Results";
      }
    }

    // Reset all votes
    function resetVotes() {
      if (confirm("Are you sure you want to reset all votes?")) {
        votes = { Candidate1: 0, Candidate2: 0, Candidate3: 0, Candidate4: 0 };
        localStorage.setItem("votes", JSON.stringify(votes));
        updateResults();
        alert("All votes have been reset to 0!");
      }
    }

    // On load keep results hidden
    document.getElementById("resultsBox").style.display = "none";
